#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int nums[n];
    for(int i = 0; i < n; i++) {
        scanf("%d", &nums[i]);
    }

    int k = 0;  // position for next non-zero

    // Move non-zero elements forward
    for(int i = 0; i < n; i++) {
        if(nums[i] != 0) {
            nums[k++] = nums[i];
        }
    }

    // Fill remaining with zeros
    while(k < n) {
        nums[k++] = 0;
    }

    // Output
    for(int i = 0; i < n; i++) {
        printf("%d ", nums[i]);
    }

    return 0;
}
