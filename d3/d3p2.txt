#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    for(int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int N = n + 1;
    int xor1 = 0, xor2 = 0;

    for(int i = 1; i <= N; i++) {
        xor1 ^= i;
    }

    for(int i = 0; i < n; i++) {
        xor2 ^= arr[i];
    }

    printf("%d", xor1 ^ xor2);

    return 0;
}
