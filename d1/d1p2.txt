#include <stdio.h>
#include <stdlib.h>

#define SIZE 100000

typedef struct Node {
    int key;    // number
    int index;  // its index
    struct Node* next;
} Node;

Node* hash[SIZE];

// simple hash function
int hashFunc(int key) {
    if (key < 0) key = -key;
    return key % SIZE;
}

// insert into hash table
void insert(int key, int index) {
    int h = hashFunc(key);
    Node* newNode = (Node*)malloc(sizeof(Node));
    newNode->key = key;
    newNode->index = index;
    newNode->next = hash[h];
    hash[h] = newNode;
}

// search in hash table
int search(int key) {
    int h = hashFunc(key);
    Node* temp = hash[h];
    while (temp) {
        if (temp->key == key)
            return temp->index;
        temp = temp->next;
    }
    return -1;
}

int main() {
    int n, target;
    scanf("%d", &n);

    int nums[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &nums[i]);
    }

    scanf("%d", &target);

    for (int i = 0; i < n; i++) {
        int complement = target - nums[i];
        int idx = search(complement);

        if (idx != -1) {
            printf("%d %d", idx, i);
            return 0;
        }

        insert(nums[i], i);
    }

    return 0;
}
