#include <stdio.h>

long long power(long long a, int b) {
    if (b == 0) return 1;

    long long half = power(a, b / 2);

    if (b % 2 == 0)
        return half * half;
    else
        return a * half * half;
}

int main() {
    long long a;
    int b;

    scanf("%lld %d", &a, &b);
    printf("%lld", power(a, b));

    return 0;
}
