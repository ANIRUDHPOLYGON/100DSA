#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

int max(int a, int b) {
    return a > b ? a : b;
}

int min(int a, int b) {
    return a < b ? a : b;
}

int maximumGap(int nums[], int n) {
    if (n < 2)
        return 0;

    int minVal = INT_MAX, maxVal = INT_MIN;

    for (int i = 0; i < n; i++) {
        minVal = min(minVal, nums[i]);
        maxVal = max(maxVal, nums[i]);
    }

    if (minVal == maxVal)
        return 0;

    int bucketSize = (maxVal - minVal) / (n - 1);
    if (bucketSize == 0)
        bucketSize = 1;

    int bucketCount = (maxVal - minVal) / bucketSize + 1;

    int *bMin = (int*)malloc(bucketCount * sizeof(int));
    int *bMax = (int*)malloc(bucketCount * sizeof(int));
    int *used = (int*)calloc(bucketCount, sizeof(int));

    for (int i = 0; i < bucketCount; i++) {
        bMin[i] = INT_MAX;
        bMax[i] = INT_MIN;
    }

    for (int i = 0; i < n; i++) {
        int idx = (nums[i] - minVal) / bucketSize;

        bMin[idx] = min(bMin[idx], nums[i]);
        bMax[idx] = max(bMax[idx], nums[i]);
        used[idx] = 1;
    }

    int prevMax = minVal;
    int ans = 0;

    for (int i = 0; i < bucketCount; i++) {
        if (!used[i])
            continue;

        ans = max(ans, bMin[i] - prevMax);
        prevMax = bMax[i];
    }

    free(bMin);
    free(bMax);
    free(used);

    return ans;
}

int main() {
    int n;
    scanf("%d", &n);

    int nums[n];
    for (int i = 0; i < n; i++)
        scanf("%d", &nums[i]);

    printf("%d", maximumGap(nums, n));

    return 0;
}
