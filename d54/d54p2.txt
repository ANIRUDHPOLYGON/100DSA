#include <stdio.h>
#include <stdlib.h>

struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
};

/* queue */
#define MAX 2000
struct TreeNode* queue[MAX];
int front = 0, rear = 0;

void push(struct TreeNode* node) {
    queue[rear++] = node;
}

struct TreeNode* pop() {
    return queue[front++];
}

int empty() {
    return front == rear;
}

/* result storage */
int result[2000][2000];
int colSize[2000];
int row = 0;

int** zigzagLevelOrder(struct TreeNode* root, int* returnSize, int** returnColumnSizes) {
    if (!root) {
        *returnSize = 0;
        return NULL;
    }

    push(root);
    int leftToRight = 1;

    while (!empty()) {
        int size = rear - front;
        colSize[row] = size;

        int temp[size];

        for (int i = 0; i < size; i++) {
            struct TreeNode* node = pop();
            temp[i] = node->val;

            if (node->left) push(node->left);
            if (node->right) push(node->right);
        }

        if (leftToRight) {
            for (int i = 0; i < size; i++)
                result[row][i] = temp[i];
        } else {
            for (int i = 0; i < size; i++)
                result[row][i] = temp[size - 1 - i];
        }

        leftToRight = !leftToRight;
        row++;
    }

    *returnSize = row;
    *returnColumnSizes = colSize;
    return result;
}
