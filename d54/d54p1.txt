#include <stdio.h>
#include <stdlib.h>

struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
};

#define MAX 2000

struct TreeNode* queue[MAX];
int front = 0, rear = 0;

void push(struct TreeNode* node) {
    queue[rear++] = node;
}

struct TreeNode* pop() {
    return queue[front++];
}

int empty() {
    return front == rear;
}

void zigzagTraversal(struct TreeNode* root) {
    if (!root) return;

    push(root);
    int leftToRight = 1;

    while (!empty()) {
        int size = rear - front;
        int level[size];

        for (int i = 0; i < size; i++) {
            struct TreeNode* node = pop();
            level[i] = node->val;

            if (node->left) push(node->left);
            if (node->right) push(node->right);
        }

        if (leftToRight) {
            for (int i = 0; i < size; i++)
                printf("%d ", level[i]);
        } else {
            for (int i = size - 1; i >= 0; i--)
                printf("%d ", level[i]);
        }

        leftToRight = !leftToRight;
    }
}
