#include <stdio.h>

int intSqrt(int n) {
    if (n == 0 || n == 1)
        return n;

    long long l = 0, r = n, ans = 0;

    while (l <= r) {
        long long mid = l + (r - l) / 2;
        long long sq = mid * mid;

        if (sq == n)
            return mid;

        if (sq < n) {
            ans = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }

    return ans;
}

int main() {
    int n;
    scanf("%d", &n);

    printf("%d", intSqrt(n));

    return 0;
}
