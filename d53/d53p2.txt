#include <stdio.h>
#include <stdlib.h>

struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
};

#define MAX 2000

struct TreeNode* queue[MAX];
int front = 0, rear = 0;

void push(struct TreeNode* node) {
    queue[rear++] = node;
}

struct TreeNode* pop() {
    return queue[front++];
}

int empty() {
    return front == rear;
}

/* result storage */
int result[2000][2000];
int colSize[2000];
int row = 0;

int** levelOrder(struct TreeNode* root, int* returnSize, int** returnColumnSizes) {
    if (!root) {
        *returnSize = 0;
        return NULL;
    }

    push(root);

    while (!empty()) {
        int size = rear - front;
        colSize[row] = size;

        for (int i = 0; i < size; i++) {
            struct TreeNode* node = pop();
            result[row][i] = node->val;

            if (node->left) push(node->left);
            if (node->right) push(node->right);
        }

        row++;
    }

    *returnSize = row;
    *returnColumnSizes = colSize;

    return result;
}
