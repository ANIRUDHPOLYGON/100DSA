#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* next;
};

// create node
struct Node* createNode(int val) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = val;
    newNode->next = NULL;
    return newNode;
}

// merge two sorted lists
struct Node* merge(struct Node* l1, struct Node* l2) {
    struct Node dummy;
    struct Node* tail = &dummy;
    dummy.next = NULL;

    while(l1 != NULL && l2 != NULL) {
        if(l1->data <= l2->data) {
            tail->next = l1;
            l1 = l1->next;
        } else {
            tail->next = l2;
            l2 = l2->next;
        }
        tail = tail->next;
    }

    if(l1 != NULL) tail->next = l1;
    if(l2 != NULL) tail->next = l2;

    return dummy.next;
}

int main() {
    int n, m;

    scanf("%d", &n);
    struct Node *head1 = NULL, *t1 = NULL;

    for(int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);
        struct Node* newNode = createNode(x);

        if(head1 == NULL) {
            head1 = newNode;
            t1 = newNode;
        } else {
            t1->next = newNode;
            t1 = newNode;
        }
    }

    scanf("%d", &m);
    struct Node *head2 = NULL, *t2 = NULL;

    for(int i = 0; i < m; i++) {
        int x;
        scanf("%d", &x);
        struct Node* newNode = createNode(x);

        if(head2 == NULL) {
            head2 = newNode;
            t2 = newNode;
        } else {
            t2->next = newNode;
            t2 = newNode;
        }
    }

    struct Node* merged = merge(head1, head2);

    while(merged != NULL) {
        printf("%d ", merged->data);
        merged = merged->next;
    }

    return 0;
}
