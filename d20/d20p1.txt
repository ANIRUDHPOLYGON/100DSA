#include <stdio.h>

#define MAX 200000

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    for(int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    // Using prefix sum frequency (shifted index)
    int freq[2 * MAX + 1] = {0};

    int prefix = 0;
    long long count = 0;

    // empty prefix sum = 0
    freq[MAX] = 1;

    for(int i = 0; i < n; i++) {
        prefix += arr[i];

        count += freq[prefix + MAX];
        freq[prefix + MAX]++;
    }

    printf("%lld", count);

    return 0;
}
