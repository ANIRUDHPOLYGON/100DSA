#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int queue[10000], stack[10000];
    int front = 0, rear = -1, top = -1;

    // enqueue input
    for (int i = 0; i < n; i++) {
        scanf("%d", &queue[++rear]);
    }

    // push queue → stack
    for (int i = front; i <= rear; i++) {
        stack[++top] = queue[i];
    }

    // pop stack → queue (reversed)
    for (int i = front; i <= rear; i++) {
        queue[i] = stack[top--];
    }

    // output
    for (int i = front; i <= rear; i++) {
        printf("%d ", queue[i]);
    }

    return 0;
}
