#include <stdio.h>

int visited[200];

void dfs(int n, int isConnected[][200], int node) {
    visited[node] = 1;

    for (int i = 0; i < n; i++) {
        if (isConnected[node][i] == 1 && !visited[i]) {
            dfs(n, isConnected, i);
        }
    }
}

int findCircleNum(int** isConnected, int n) {
    for (int i = 0; i < n; i++)
        visited[i] = 0;

    int provinces = 0;

    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            dfs(n, isConnected, i);
            provinces++;
        }
    }

    return provinces;
}
