#include <stdio.h>

int dir[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};

void dfs(int** image, int m, int n, int r, int c, int orig, int color) {
    if (r < 0 || c < 0 || r >= m || c >= n) return;
    if (image[r][c] != orig) return;

    image[r][c] = color;

    for (int i = 0; i < 4; i++) {
        dfs(image, m, n, r + dir[i][0], c + dir[i][1], orig, color);
    }
}

int** floodFill(int** image, int imageSize, int* imageColSize,
                int sr, int sc, int color, int* returnSize, int** returnColumnSizes) {

    int orig = image[sr][sc];

    if (orig == color) return image;

    dfs(image, imageSize, imageColSize[0], sr, sc, orig, color);

    *returnSize = imageSize;
    return image;
}
