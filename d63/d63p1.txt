#include <stdio.h>
#include <stdlib.h>

#define MAX 1000

int visited[MAX];

void dfs(int node, int** adj, int* adjSize) {
    visited[node] = 1;
    printf("%d ", node);

    for (int i = 0; i < adjSize[node]; i++) {
        int nei = adj[node][i];
        if (!visited[nei]) {
            dfs(nei, adj, adjSize);
        }
    }
}

void startDFS(int n, int** adj, int* adjSize, int s) {
    for (int i = 0; i < n; i++)
        visited[i] = 0;

    dfs(s, adj, adjSize);
}
