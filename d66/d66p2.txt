#include <stdio.h>
#include <stdlib.h>

int visited[2000];
int recStack[2000];

int dfs(int node, int** adj, int* adjSize) {
    visited[node] = 1;
    recStack[node] = 1;

    for (int i = 0; i < adjSize[node]; i++) {
        int nei = adj[node][i];

        if (!visited[nei]) {
            if (dfs(nei, adj, adjSize))
                return 1;
        }
        else if (recStack[nei]) {
            return 1; // cycle found
        }
    }

    recStack[node] = 0;
    return 0;
}

int canFinish(int numCourses, int** prerequisites, int prerequisitesSize, int* prerequisitesColSize) {

    // adjacency list
    int** adj = (int**)malloc(numCourses * sizeof(int*));
    int* adjSize = (int*)calloc(numCourses, sizeof(int));

    for (int i = 0; i < numCourses; i++) {
        adj[i] = (int*)malloc(5000 * sizeof(int));
    }

    // build graph
    for (int i = 0; i < prerequisitesSize; i++) {
        int a = prerequisites[i][0];
        int b = prerequisites[i][1];

        adj[b][adjSize[b]++] = a;
    }

    for (int i = 0; i < numCourses; i++) {
        visited[i] = 0;
        recStack[i] = 0;
    }

    // check all components
    for (int i = 0; i < numCourses; i++) {
        if (!visited[i]) {
            if (dfs(i, adj, adjSize))
                return 0; // cycle → cannot finish
        }
    }

    return 1;
}
