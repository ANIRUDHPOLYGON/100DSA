#include <stdio.h>

#define MAX 1000

int visited[MAX];
int recStack[MAX];

int dfs(int node, int** adj, int* adjSize) {
    visited[node] = 1;
    recStack[node] = 1;

    for (int i = 0; i < adjSize[node]; i++) {
        int nei = adj[node][i];

        if (!visited[nei]) {
            if (dfs(nei, adj, adjSize))
                return 1;
        }
        else if (recStack[nei]) {
            return 1; // back edge → cycle
        }
    }

    recStack[node] = 0;
    return 0;
}

int hasCycle(int V, int** adj, int* adjSize) {
    for (int i = 0; i < V; i++) {
        visited[i] = 0;
        recStack[i] = 0;
    }

    for (int i = 0; i < V; i++) {
        if (!visited[i]) {
            if (dfs(i, adj, adjSize))
                return 1;
        }
    }

    return 0;
}
