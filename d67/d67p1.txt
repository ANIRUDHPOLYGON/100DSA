#include <stdio.h>
#include <stdlib.h>

#define MAX 1000

int visited[MAX];
int stack[MAX];
int top = -1;

void dfs(int node, int** adj, int* adjSize) {
    visited[node] = 1;

    for (int i = 0; i < adjSize[node]; i++) {
        int nei = adj[node][i];

        if (!visited[nei]) {
            dfs(nei, adj, adjSize);
        }
    }

    // push after visiting all children
    stack[++top] = node;
}

void topoSort(int V, int** adj, int* adjSize) {
    for (int i = 0; i < V; i++)
        visited[i] = 0;

    for (int i = 0; i < V; i++) {
        if (!visited[i]) {
            dfs(i, adj, adjSize);
        }
    }

    // print reverse of finishing order
    for (int i = top; i >= 0; i--) {
        printf("%d ", stack[i]);
    }
}
