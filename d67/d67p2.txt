#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

int visited[2000];
int recStack[2000];
int stack[2000];
int top = -1;
int hasCycle = 0;

void dfs(int node, int** adj, int* adjSize) {
    visited[node] = 1;
    recStack[node] = 1;

    for (int i = 0; i < adjSize[node]; i++) {
        int nei = adj[node][i];

        if (!visited[nei]) {
            dfs(nei, adj, adjSize);
        }
        else if (recStack[nei]) {
            hasCycle = 1;
            return;
        }

        if (hasCycle) return;
    }

    recStack[node] = 0;
    stack[++top] = node; // postorder push
}

int* findOrder(int numCourses, int** prerequisites, int prerequisitesSize,
               int* prerequisitesColSize, int* returnSize) {

    int** adj = (int**)malloc(numCourses * sizeof(int*));
    int* adjSize = (int*)calloc(numCourses, sizeof(int));

    for (int i = 0; i < numCourses; i++)
        adj[i] = (int*)malloc(1000 * sizeof(int));

    // build graph
    for (int i = 0; i < prerequisitesSize; i++) {
        int a = prerequisites[i][0];
        int b = prerequisites[i][1];
        adj[b][adjSize[b]++] = a;
    }

    for (int i = 0; i < numCourses; i++) {
        visited[i] = 0;
        recStack[i] = 0;
    }

    top = -1;
    hasCycle = 0;

    for (int i = 0; i < numCourses; i++) {
        if (!visited[i]) {
            dfs(i, adj, adjSize);
        }
    }

    if (hasCycle) {
        *returnSize = 0;
        return NULL;
    }

    int* res = (int*)malloc(numCourses * sizeof(int));

    for (int i = 0; i < numCourses; i++) {
        res[i] = stack[top--];
    }

    *returnSize = numCourses;
    return res;
}
