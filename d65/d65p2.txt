#include <stdio.h>
#include <stdlib.h>

#define MAX 100000

int visited[MAX];

int dfs(int node, int parent, int** adj, int* adjSize) {
    visited[node] = 1;

    for (int i = 0; i < adjSize[node]; i++) {
        int nei = adj[node][i];

        if (!visited[nei]) {
            if (dfs(nei, node, adj, adjSize))
                return 1;
        }
        else if (nei != parent) {
            return 1; // cycle found
        }
    }

    return 0;
}

int isCycle(int V, int** adj, int* adjSize) {
    for (int i = 0; i < V; i++)
        visited[i] = 0;

    for (int i = 0; i < V; i++) {
        if (!visited[i]) {
            if (dfs(i, -1, adj, adjSize))
                return 1;
        }
    }

    return 0;
}
