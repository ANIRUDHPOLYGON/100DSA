#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Node {
    int val;
    struct Node* next;
};

struct Node* top = NULL;

// push
void push(int x) {
    struct Node* node = (struct Node*)malloc(sizeof(struct Node));
    node->val = x;
    node->next = top;
    top = node;
}

// pop
int pop() {
    if(top == NULL) return 0;

    struct Node* temp = top;
    int val = temp->val;
    top = top->next;
    free(temp);
    return val;
}

// check operator
int isOp(char* s) {
    return (strlen(s) == 1 &&
           (s[0] == '+' || s[0] == '-' ||
            s[0] == '*' || s[0] == '/'));
}

int main() {
    char token[20];

    while(scanf("%s", token) != EOF) {

        if(isOp(token)) {
            int b = pop();
            int a = pop();
            int res;

            switch(token[0]) {
                case '+': res = a + b; break;
                case '-': res = a - b; break;
                case '*': res = a * b; break;
                case '/': res = a / b; break;
            }

            push(res);
        }
        else {
            push(atoi(token));
        }
    }

    printf("%d\n", pop());

    return 0;
}
