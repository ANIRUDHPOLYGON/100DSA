#include <stdio.h>
#include <limits.h>

#define MAX 1000

int visited[MAX];
int dist[MAX];

int minDistance(int n) {
    int min = INT_MAX, idx = -1;

    for (int i = 0; i < n; i++) {
        if (!visited[i] && dist[i] < min) {
            min = dist[i];
            idx = i;
        }
    }

    return idx;
}

void dijkstra(int n, int adj[MAX][MAX], int src) {

    for (int i = 0; i < n; i++) {
        dist[i] = INT_MAX;
        visited[i] = 0;
    }

    dist[src] = 0;

    for (int i = 0; i < n - 1; i++) {

        int u = minDistance(n);
        visited[u] = 1;

        for (int v = 0; v < n; v++) {
            if (!visited[v] &&
                adj[u][v] &&
                dist[u] != INT_MAX &&
                dist[u] + adj[u][v] < dist[v]) {

                dist[v] = dist[u] + adj[u][v];
            }
        }
    }

    // print distances
    for (int i = 0; i < n; i++) {
        printf("%d ", dist[i]);
    }
}
