#include <stdio.h>
#include <limits.h>

#define MAX 105

int visited[MAX];
int dist[MAX];

int minDistance(int n) {
    int min = INT_MAX, idx = -1;

    for (int i = 1; i <= n; i++) {
        if (!visited[i] && dist[i] < min) {
            min = dist[i];
            idx = i;
        }
    }

    return idx;
}

int networkDelayTime(int** times, int timesSize, int* timesColSize, int n, int k) {

    int adj[MAX][MAX] = {0};

    // build graph (take min if multiple edges conceptually existed)
    for (int i = 0; i < timesSize; i++) {
        int u = times[i][0];
        int v = times[i][1];
        int w = times[i][2];

        adj[u][v] = w;
    }

    for (int i = 1; i <= n; i++) {
        dist[i] = INT_MAX;
        visited[i] = 0;
    }

    dist[k] = 0;

    for (int i = 1; i <= n; i++) {

        int u = minDistance(n);
        if (u == -1) break;

        visited[u] = 1;

        for (int v = 1; v <= n; v++) {
            if (!visited[v] &&
                adj[u][v] &&
                dist[u] != INT_MAX &&
                dist[u] + adj[u][v] < dist[v]) {

                dist[v] = dist[u] + adj[u][v];
            }
        }
    }

    int ans = 0;

    for (int i = 1; i <= n; i++) {
        if (dist[i] == INT_MAX)
            return -1;

        if (dist[i] > ans)
            ans = dist[i];
    }

    return ans;
}
