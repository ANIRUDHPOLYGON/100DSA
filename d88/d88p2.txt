#include <stdio.h>

void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}

void sort(int a[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (a[j] > a[j + 1])
                swap(&a[j], &a[j + 1]);
        }
    }
}

int canPlace(int pos[], int n, int m, int dist) {
    int count = 1;
    int last = pos[0];

    for (int i = 1; i < n; i++) {
        if (pos[i] - last >= dist) {
            count++;
            last = pos[i];
        }
        if (count >= m)
            return 1;
    }

    return 0;
}

int maxDistance(int pos[], int n, int m) {
    sort(pos, n);

    int low = 1;
    int high = pos[n - 1] - pos[0];
    int ans = 0;

    while (low <= high) {
        int mid = low + (high - low) / 2;

        if (canPlace(pos, n, m, mid)) {
            ans = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    return ans;
}

int main() {
    int n, m;
    scanf("%d", &n);

    int pos[n];
    for (int i = 0; i < n; i++)
        scanf("%d", &pos[i]);

    scanf("%d", &m);

    printf("%d", maxDistance(pos, n, m));

    return 0;
}
