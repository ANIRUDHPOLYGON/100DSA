#include <stdio.h>

#define MAX 100

int dir[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};

int orangesRotting(int grid[MAX][MAX], int m, int n) {
    int queue[MAX * MAX][2];
    int front = 0, rear = 0;

    int fresh = 0;

    // initialize queue with all rotten oranges
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (grid[i][j] == 2) {
                queue[rear][0] = i;
                queue[rear][1] = j;
                rear++;
            }
            else if (grid[i][j] == 1) {
                fresh++;
            }
        }
    }

    int minutes = 0;

    while (front < rear && fresh > 0) {
        int size = rear - front;

        for (int i = 0; i < size; i++) {
            int r = queue[front][0];
            int c = queue[front][1];
            front++;

            for (int d = 0; d < 4; d++) {
                int nr = r + dir[d][0];
                int nc = c + dir[d][1];

                if (nr >= 0 && nc >= 0 && nr < m && nc < n &&
                    grid[nr][nc] == 1) {

                    grid[nr][nc] = 2;
                    fresh--;

                    queue[rear][0] = nr;
                    queue[rear][1] = nc;
                    rear++;
                }
            }
        }

        minutes++;
    }

    return (fresh == 0) ? minutes : -1;
}
