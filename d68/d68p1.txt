#include <stdio.h>
#include <stdlib.h>

int* topoSort(int V, int** adj, int* adjSize, int* returnSize) {

    int* indegree = (int*)calloc(V, sizeof(int));

    // compute indegree
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < adjSize[i]; j++) {
            int v = adj[i][j];
            indegree[v]++;
        }
    }

    int* queue = (int*)malloc(V * sizeof(int));
    int front = 0, rear = 0;

    // push all 0 indegree nodes
    for (int i = 0; i < V; i++) {
        if (indegree[i] == 0)
            queue[rear++] = i;
    }

    int* result = (int*)malloc(V * sizeof(int));
    int idx = 0;

    while (front < rear) {
        int node = queue[front++];
        result[idx++] = node;

        for (int i = 0; i < adjSize[node]; i++) {
            int nei = adj[node][i];

            indegree[nei]--;

            if (indegree[nei] == 0) {
                queue[rear++] = nei;
            }
        }
    }

    // cycle check
    if (idx != V) {
        *returnSize = 0;
        return NULL;
    }

    *returnSize = V;
    return result;
}
