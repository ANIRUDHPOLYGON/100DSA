#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define ALPHABET 26

char* alienOrder(char** words, int wordsSize) {

    int adj[ALPHABET][ALPHABET] = {0};
    int indegree[ALPHABET] = {0};
    int present[ALPHABET] = {0};

    // mark present letters
    for (int i = 0; i < wordsSize; i++) {
        for (int j = 0; words[i][j]; j++) {
            present[words[i][j] - 'a'] = 1;
        }
    }

    // build graph
    for (int i = 0; i < wordsSize - 1; i++) {
        char* w1 = words[i];
        char* w2 = words[i + 1];

        int len1 = strlen(w1);
        int len2 = strlen(w2);
        int minLen = len1 < len2 ? len1 : len2;

        int found = 0;

        for (int j = 0; j < minLen; j++) {
            if (w1[j] != w2[j]) {
                int u = w1[j] - 'a';
                int v = w2[j] - 'a';

                if (!adj[u][v]) {
                    adj[u][v] = 1;
                    indegree[v]++;
                }

                found = 1;
                break;
            }
        }

        // invalid case: prefix issue
        if (!found && len1 > len2)
            return "";
    }

    // Kahn BFS
    int queue[ALPHABET];
    int front = 0, rear = 0;

    for (int i = 0; i < ALPHABET; i++) {
        if (present[i] && indegree[i] == 0)
            queue[rear++] = i;
    }

    char* result = (char*)malloc(ALPHABET + 1);
    int idx = 0;

    while (front < rear) {
        int node = queue[front++];

        result[idx++] = node + 'a';

        for (int i = 0; i < ALPHABET; i++) {
            if (adj[node][i]) {
                indegree[i]--;
                if (indegree[i] == 0)
                    queue[rear++] = i;
            }
        }
    }

    // cycle check
    int total = 0;
    for (int i = 0; i < ALPHABET; i++) {
        if (present[i]) total++;
    }

    if (idx != total)
        return "";

    result[idx] = '\0';
    return result;
}
