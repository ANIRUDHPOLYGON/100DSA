#include <stdio.h>
#include <stdbool.h>

#define MAX 1000

int adj[MAX][MAX];
bool visited[MAX];

void dfs(int node, int n) {
    visited[node] = true;

    for (int i = 1; i <= n; i++) {
        if (adj[node][i] && !visited[i]) {
            dfs(i, n);
        }
    }
}

int main() {
    int n, m;
    scanf("%d %d", &n, &m);

    // initialize
    for (int i = 1; i <= n; i++) {
        visited[i] = false;
        for (int j = 1; j <= n; j++) {
            adj[i][j] = 0;
        }
    }

    // build graph
    for (int i = 0; i < m; i++) {
        int u, v;
        scanf("%d %d", &u, &v);

        adj[u][v] = 1;
        adj[v][u] = 1;
    }

    int components = 0;

    for (int i = 1; i <= n; i++) {
        if (!visited[i]) {
            components++;
            dfs(i, n);
        }
    }

    printf("%d", components);

    return 0;
}
