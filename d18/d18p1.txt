#include <stdio.h>

void reverse(int arr[], int l, int r) {
    while(l < r) {
        int temp = arr[l];
        arr[l] = arr[r];
        arr[r] = temp;
        l++;
        r--;
    }
}

int main() {
    int n, k;
    scanf("%d", &n);

    int arr[n];
    for(int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    scanf("%d", &k);

    k = k % n;

    // Step 1: reverse full array
    reverse(arr, 0, n - 1);

    // Step 2: reverse first k elements
    reverse(arr, 0, k - 1);

    // Step 3: reverse remaining
    reverse(arr, k, n - 1);

    // Output
    for(int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }

    return 0;
}
