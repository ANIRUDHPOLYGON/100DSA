#include <stdio.h>
#include <stdbool.h>

int visited[1000];

void dfs(int** rooms, int* roomsColSize, int node) {
    visited[node] = 1;

    for (int i = 0; i < roomsColSize[node]; i++) {
        int key = rooms[node][i];
        if (!visited[key]) {
            dfs(rooms, roomsColSize, key);
        }
    }
}

bool canVisitAllRooms(int** rooms, int roomsSize, int* roomsColSize) {
    for (int i = 0; i < roomsSize; i++)
        visited[i] = 0;

    dfs(rooms, roomsColSize, 0);

    for (int i = 0; i < roomsSize; i++) {
        if (!visited[i])
            return false;
    }

    return true;
}
