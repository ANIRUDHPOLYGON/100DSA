#include <stdio.h>
#include <stdlib.h>

struct Node {
    int coeff;
    int exp;
    struct Node* next;
};

struct Node* createNode(int c, int e) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->coeff = c;
    newNode->exp = e;
    newNode->next = NULL;
    return newNode;
}

int main() {
    int n;
    scanf("%d", &n);

    struct Node *head = NULL, *tail = NULL;

    // build polynomial linked list
    for(int i = 0; i < n; i++) {
        int c, e;
        scanf("%d %d", &c, &e);

        struct Node* node = createNode(c, e);

        if(head == NULL) {
            head = node;
            tail = node;
        } else {
            tail->next = node;
            tail = node;
        }
    }

    // print polynomial
    struct Node* temp = head;

    while(temp != NULL) {
        printf("%dx^%d", temp->coeff, temp->exp);

        if(temp->next != NULL) {
            printf(" + ");
        }

        temp = temp->next;
    }

    return 0;
}
