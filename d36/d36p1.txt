#include <stdio.h>

#define MAX 1000

int queue[MAX];
int front = 0;
int rear = -1;

void enqueue(int x) {
    queue[++rear] = x;
}

void display(int n) {
    for(int i = front; i <= rear; i++) {
        printf("%d ", queue[i]);
    }
}

int main() {
    int n;
    scanf("%d", &n);

    for(int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);
        enqueue(x);
    }

    display(n);

    return 0;
}
