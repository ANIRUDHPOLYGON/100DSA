#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* next;
};

struct Node* createNode(int val) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = val;
    newNode->next = NULL;
    return newNode;
}

int main() {
    int n, val;
    scanf("%d", &n);

    struct Node *head = NULL, *temp = NULL;

    // create linked list
    for(int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);

        struct Node* newNode = createNode(x);

        if(head == NULL) {
            head = newNode;
            temp = newNode;
        } else {
            temp->next = newNode;
            temp = newNode;
        }
    }

    scanf("%d", &val);

    // remove from head while needed
    while(head != NULL && head->data == val) {
        struct Node* del = head;
        head = head->next;
        free(del);
    }

    // remove remaining nodes
    struct Node* curr = head;

    while(curr != NULL && curr->next != NULL) {
        if(curr->next->data == val) {
            struct Node* del = curr->next;
            curr->next = curr->next->next;
            free(del);
        } else {
            curr = curr->next;
        }
    }

    // print result
    temp = head;
    while(temp != NULL) {
        printf("%d ", temp->data);
        temp = temp->next;
    }

    return 0;
}
