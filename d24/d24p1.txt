#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* next;
};

struct Node* createNode(int val) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = val;
    newNode->next = NULL;
    return newNode;
}

int main() {
    int n, key;
    scanf("%d", &n);

    struct Node *head = NULL, *temp = NULL;

    // create list
    for(int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);

        struct Node* newNode = createNode(x);

        if(head == NULL) {
            head = newNode;
            temp = newNode;
        } else {
            temp->next = newNode;
            temp = newNode;
        }
    }

    scanf("%d", &key);

    struct Node *curr = head, *prev = NULL;

    // if head needs deletion
    if(curr != NULL && curr->data == key) {
        head = curr->next;
        free(curr);
    } else {
        while(curr != NULL && curr->data != key) {
            prev = curr;
            curr = curr->next;
        }

        if(curr != NULL) {
            prev->next = curr->next;
            free(curr);
        }
    }

    // print list
    temp = head;
    while(temp != NULL) {
        printf("%d ", temp->data);
        temp = temp->next;
    }

    return 0;
}
