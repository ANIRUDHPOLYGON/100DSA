#include <stdio.h>
#include <stdlib.h>

int cmp(const void *a, const void *b) {
    int *x = (int *)a;
    int *y = (int *)b;
    return x[0] - y[0];
}

int minRooms(int intervals[][2], int n) {
    int start[n], end[n];

    for (int i = 0; i < n; i++) {
        start[i] = intervals[i][0];
        end[i] = intervals[i][1];
    }

    qsort(start, n, sizeof(int), (__compar_fn_t)cmp);
    qsort(end, n, sizeof(int), (__compar_fn_t)cmp);

    int rooms = 0, j = 0;

    for (int i = 0; i < n; i++) {
        if (start[i] < end[j])
            rooms++;
        else
            j++;
    }

    return rooms;
}
