#include <stdio.h>

#define MAX 1000

int pq[MAX];
int size = 0;

// insert in sorted order
void insert(int x) {
    int i = size - 1;

    // shift larger elements right
    while(i >= 0 && pq[i] > x) {
        pq[i + 1] = pq[i];
        i--;
    }

    pq[i + 1] = x;
    size++;
}

// delete highest priority (smallest)
int delete() {
    if(size == 0) return -1;

    int val = pq[0];

    for(int i = 1; i < size; i++) {
        pq[i - 1] = pq[i];
    }

    size--;
    return val;
}

// peek
int peek() {
    if(size == 0) return -1;
    return pq[0];
}

int main() {
    int n;
    scanf("%d", &n);

    for(int i = 0; i < n; i++) {
        char op[10];
        scanf("%s", op);

        if(op[0] == 'i') {
            int x;
            scanf("%d", &x);
            insert(x);
        }
        else if(op[0] == 'd') {
            printf("%d\n", delete());
        }
        else if(op[0] == 'p') {
            printf("%d\n", peek());
        }
    }

    return 0;
}
