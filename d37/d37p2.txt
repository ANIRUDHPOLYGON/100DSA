#include <stdio.h>

#define MAX 10005

typedef struct {
    int heap[MAX];
    int size;
    int k;
} KthLargest;

// swap
void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}

// heapify up
void up(KthLargest* obj, int i) {
    while(i > 0) {
        int parent = (i - 1) / 2;
        if(obj->heap[parent] > obj->heap[i]) {
            swap(&obj->heap[parent], &obj->heap[i]);
            i = parent;
        } else break;
    }
}

// heapify down
void down(KthLargest* obj, int i) {
    int smallest = i;
    int left = 2*i + 1;
    int right = 2*i + 2;

    if(left < obj->size && obj->heap[left] < obj->heap[smallest])
        smallest = left;

    if(right < obj->size && obj->heap[right] < obj->heap[smallest])
        smallest = right;

    if(smallest != i) {
        swap(&obj->heap[i], &obj->heap[smallest]);
        down(obj, smallest);
    }
}

// create
KthLargest* create(int k, int* nums, int numsSize) {
    static KthLargest obj;
    obj.size = 0;
    obj.k = k;

    for(int i = 0; i < numsSize; i++) {
        add(&obj, nums[i]);
    }

    return &obj;
}

// add
int add(KthLargest* obj, int val) {
    obj->heap[obj->size++] = val;
    up(obj, obj->size - 1);

    if(obj->size > obj->k) {
        obj->heap[0] = obj->heap[obj->size - 1];
        obj->size--;
        down(obj, 0);
    }

    return obj->heap[0];
}
