#include <stdio.h>

int findPeak(int nums[], int n) {
    int l = 0, r = n - 1;

    while (l < r) {
        int mid = l + (r - l) / 2;

        if (nums[mid] < nums[mid + 1])
            l = mid + 1;
        else
            r = mid;
    }

    return l;
}

int main() {
    int n;
    scanf("%d", &n);

    int nums[n];
    for (int i = 0; i < n; i++)
        scanf("%d", &nums[i]);

    printf("%d", findPeak(nums, n));

    return 0;
}
