#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node *left, *right;
} Node;

Node* createNode(int x) {
    Node* n = (Node*)malloc(sizeof(Node));
    n->data = x;
    n->left = n->right = NULL;
    return n;
}

/* Queue for building + traversal */
Node* queue[10000];
int front = 0, rear = -1;

void push(Node* node) {
    queue[++rear] = node;
}

Node* pop() {
    return queue[front++];
}

int isEmpty() {
    return front > rear;
}

/* Build tree from level order (-1 = NULL) */
Node* buildTree(int arr[], int n) {
    if (n == 0) return NULL;

    Node* root = createNode(arr[0]);
    push(root);

    int i = 1;

    while (i < n) {
        Node* curr = pop();

        // left
        if (i < n && arr[i] != -1) {
            curr->left = createNode(arr[i]);
            push(curr->left);
        }
        i++;

        // right
        if (i < n && arr[i] != -1) {
            curr->right = createNode(arr[i]);
            push(curr->right);
        }
        i++;
    }

    return root;
}

/* Level order traversal */
void levelOrder(Node* root) {
    if (!root) return;

    front = 0;
    rear = -1;

    push(root);

    while (!isEmpty()) {
        Node* curr = pop();
        printf("%d ", curr->data);

        if (curr->left) push(curr->left);
        if (curr->right) push(curr->right);
    }
}

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    for (int i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    Node* root = buildTree(arr, n);

    levelOrder(root);

    return 0;
}
