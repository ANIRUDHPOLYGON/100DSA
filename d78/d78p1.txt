#include <stdio.h>
#include <limits.h>
#include <stdbool.h>

#define MAX 1005

int minKey(int key[], bool mstSet[], int n) {
    int min = INT_MAX, idx = -1;

    for (int i = 1; i <= n; i++) {
        if (!mstSet[i] && key[i] < min) {
            min = key[i];
            idx = i;
        }
    }

    return idx;
}

int main() {
    int n, m;
    scanf("%d %d", &n, &m);

    int graph[MAX][MAX];

    // initialize graph with INF
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            graph[i][j] = INT_MAX;
        }
    }

    for (int i = 0; i < m; i++) {
        int u, v, w;
        scanf("%d %d %d", &u, &v, &w);

        if (w < graph[u][v]) {
            graph[u][v] = w;
            graph[v][u] = w;
        }
    }

    int key[MAX];
    bool mstSet[MAX];

    for (int i = 1; i <= n; i++) {
        key[i] = INT_MAX;
        mstSet[i] = false;
    }

    key[1] = 0;

    int totalWeight = 0;

    for (int count = 1; count <= n; count++) {

        int u = minKey(key, mstSet, n);
        mstSet[u] = true;

        totalWeight += key[u];

        for (int v = 1; v <= n; v++) {
            if (graph[u][v] != INT_MAX && !mstSet[v] && graph[u][v] < key[v]) {
                key[v] = graph[u][v];
            }
        }
    }

    printf("%d", totalWeight);

    return 0;
}
