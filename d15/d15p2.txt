#include <stdio.h>

int main() {
    int m, n;
    scanf("%d %d", &m, &n);

    int a[m][n];

    for(int i = 0; i < m; i++) {
        for(int j = 0; j < n; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    int row0 = 0, col0 = 0;

    // check first row
    for(int j = 0; j < n; j++) {
        if(a[0][j] == 0) row0 = 1;
    }

    // check first column
    for(int i = 0; i < m; i++) {
        if(a[i][0] == 0) col0 = 1;
    }

    // mark using first row/col
    for(int i = 1; i < m; i++) {
        for(int j = 1; j < n; j++) {
            if(a[i][j] == 0) {
                a[i][0] = 0;
                a[0][j] = 0;
            }
        }
    }

    // set zeros based on marks
    for(int i = 1; i < m; i++) {
        for(int j = 1; j < n; j++) {
            if(a[i][0] == 0 || a[0][j] == 0) {
                a[i][j] = 0;
            }
        }
    }

    // first row
    if(row0) {
        for(int j = 0; j < n; j++) {
            a[0][j] = 0;
        }
    }

    // first column
    if(col0) {
        for(int i = 0; i < m; i++) {
            a[i][0] = 0;
        }
    }

    // output
    for(int i = 0; i < m; i++) {
        for(int j = 0; j < n; j++) {
            printf("%d ", a[i][j]);
        }
        printf("\n");
    }

    return 0;
}
