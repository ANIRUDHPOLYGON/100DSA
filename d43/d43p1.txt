#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node *left, *right;
} Node;

Node* createNode(int val) {
    Node* node = (Node*)malloc(sizeof(Node));
    node->data = val;
    node->left = node->right = NULL;
    return node;
}

// simple queue for tree nodes
Node* queue[10000];
int front = 0, rear = -1;

void push(Node* node) {
    queue[++rear] = node;
}

Node* pop() {
    return queue[front++];
}

int isEmpty() {
    return front > rear;
}

// inorder traversal
void inorder(Node* root) {
    if (!root) return;
    inorder(root->left);
    printf("%d ", root->data);
    inorder(root->right);
}

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    for (int i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    if (n == 0) return 0;

    Node* root = createNode(arr[0]);
    push(root);

    int i = 1;

    while (i < n && !isEmpty()) {
        Node* curr = pop();

        // left child
        if (i < n && arr[i] != -1) {
            curr->left = createNode(arr[i]);
            push(curr->left);
        }
        i++;

        // right child
        if (i < n && arr[i] != -1) {
            curr->right = createNode(arr[i]);
            push(curr->right);
        }
        i++;
    }

    inorder(root);

    return 0;
}
