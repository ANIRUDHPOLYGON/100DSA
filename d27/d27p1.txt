#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* next;
};

// create node
struct Node* createNode(int val) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = val;
    newNode->next = NULL;
    return newNode;
}

// length
int length(struct Node* head) {
    int len = 0;
    while(head != NULL) {
        len++;
        head = head->next;
    }
    return len;
}

// advance pointer
struct Node* advance(struct Node* head, int diff) {
    while(diff--) {
        head = head->next;
    }
    return head;
}

int main() {
    int n, m;
    scanf("%d", &n);

    struct Node *head1 = NULL, *t1 = NULL;

    for(int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);

        struct Node* node = createNode(x);

        if(head1 == NULL) {
            head1 = node;
            t1 = node;
        } else {
            t1->next = node;
            t1 = node;
        }
    }

    scanf("%d", &m);

    struct Node *head2 = NULL, *t2 = NULL;

    for(int i = 0; i < m; i++) {
        int x;
        scanf("%d", &x);

        struct Node* node = createNode(x);

        if(head2 == NULL) {
            head2 = node;
            t2 = node;
        } else {
            t2->next = node;
            t2 = node;
        }
    }

    int len1 = length(head1);
    int len2 = length(head2);

    struct Node *p1 = head1;
    struct Node *p2 = head2;

    // align
    if(len1 > len2) {
        p1 = advance(p1, len1 - len2);
    } else {
        p2 = advance(p2, len2 - len1);
    }

    // find intersection
    while(p1 != NULL && p2 != NULL) {
        if(p1 == p2) {
            printf("%d", p1->data);
            return 0;
        }
        p1 = p1->next;
        p2 = p2->next;
    }

    printf("No Intersection");

    return 0;
}
