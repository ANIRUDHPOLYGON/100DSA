#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node *left, *right;
} Node;

Node* createNode(int x) {
    Node* n = (Node*)malloc(sizeof(Node));
    n->data = x;
    n->left = n->right = NULL;
    return n;
}

/* Queue for tree construction */
Node* queue[10000];
int front = 0, rear = -1;

void push(Node* node) {
    queue[++rear] = node;
}

Node* pop() {
    return queue[front++];
}

/* Build tree from level order */
Node* buildTree(int arr[], int n) {
    if (n == 0) return NULL;

    Node* root = createNode(arr[0]);
    push(root);

    int i = 1;

    while (i < n) {
        Node* curr = pop();

        // left child
        if (i < n && arr[i] != -1) {
            curr->left = createNode(arr[i]);
            push(curr->left);
        }
        i++;

        // right child
        if (i < n && arr[i] != -1) {
            curr->right = createNode(arr[i]);
            push(curr->right);
        }
        i++;
    }

    return root;
}

/* Height calculation */
int height(Node* root) {
    if (root == NULL) return 0;

    int lh = height(root->left);
    int rh = height(root->right);

    return 1 + (lh > rh ? lh : rh);
}

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    for (int i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    Node* root = buildTree(arr, n);

    printf("%d", height(root));

    return 0;
}
