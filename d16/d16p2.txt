#include <stdio.h>

void reverse(int nums[], int l, int r) {
    while(l < r) {
        int temp = nums[l];
        nums[l] = nums[r];
        nums[r] = temp;
        l++;
        r--;
    }
}

int main() {
    int n, k;
    scanf("%d", &n);

    int nums[n];
    for(int i = 0; i < n; i++) {
        scanf("%d", &nums[i]);
    }

    scanf("%d", &k);

    k = k % n;  // handle large k

    // Step 1: reverse whole array
    reverse(nums, 0, n - 1);

    // Step 2: reverse first k elements
    reverse(nums, 0, k - 1);

    // Step 3: reverse remaining
    reverse(nums, k, n - 1);

    // Output
    for(int i = 0; i < n; i++) {
        printf("%d ", nums[i]);
    }

    return 0;
}
