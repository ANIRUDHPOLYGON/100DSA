#include <stdio.h>
#include <stdlib.h>

int main() {
    int n;
    scanf("%d", &n);

    int a[n];
    int max = 0;

    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
        if (a[i] > max)
            max = a[i];
    }

    int *freq = (int*)calloc(max + 1, sizeof(int));

    for (int i = 0; i < n; i++)
        freq[a[i]]++;

    // prefix sum
    for (int i = 1; i <= max; i++)
        freq[i] += freq[i - 1];

    int *output = (int*)malloc(n * sizeof(int));

    // build output (stable)
    for (int i = n - 1; i >= 0; i--) {
        output[freq[a[i]] - 1] = a[i];
        freq[a[i]]--;
    }

    for (int i = 0; i < n; i++)
        printf("%d ", output[i]);

    return 0;
}
