#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, m;
    scanf("%d %d", &n, &m);

    int arr1[n], arr2[m];

    for (int i = 0; i < n; i++)
        scanf("%d", &arr1[i]);

    for (int i = 0; i < m; i++)
        scanf("%d", &arr2[i]);

    int freq[1001] = {0};

    for (int i = 0; i < n; i++)
        freq[arr1[i]]++;

    int idx = 0;

    // place elements in order of arr2
    for (int i = 0; i < m; i++) {
        while (freq[arr2[i]] > 0) {
            arr1[idx++] = arr2[i];
            freq[arr2[i]]--;
        }
    }

    // remaining elements in sorted order
    for (int i = 0; i <= 1000; i++) {
        while (freq[i] > 0) {
            arr1[idx++] = i;
            freq[i]--;
        }
    }

    for (int i = 0; i < n; i++)
        printf("%d ", arr1[i]);

    return 0;
}
