#include <stdio.h>

#define MAX 30000

int stack[MAX];
int minStack[MAX];
int top = -1;

void push(int val) {
    stack[++top] = val;

    if(top == 0) {
        minStack[top] = val;
    } else {
        if(val < minStack[top - 1])
            minStack[top] = val;
        else
            minStack[top] = minStack[top - 1];
    }
}

void pop() {
    if(top >= 0) {
        top--;
    }
}

int topElement() {
    return stack[top];
}

int getMin() {
    return minStack[top];
}
